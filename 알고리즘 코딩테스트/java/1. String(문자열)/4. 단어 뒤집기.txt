package com.coding.java04;

import java.util.ArrayList;
import java.util.Scanner;

public class Main {
	
	public void solution(int num, String[] str) {
		
		for(String x : str) {
			String answer = ""; 
			for(int i = x.length() - 1; i >= 0; i--) {
				answer += x.charAt(i);
			}
			System.out.println(answer);
		}
	}
	
	
	public ArrayList<String> solution3(int num, String[] strArr){
		ArrayList<String> answer = new ArrayList<>();
		
		for(String x : strArr) {
			String tmp = new StringBuilder(x).reverse().toString();
			answer.add(tmp);
		}
		
		return answer;
	}
	
	public ArrayList<String> solution4(int n, String[] str){
		ArrayList<String> answer = new ArrayList<>();
		for(String x : str) {
			
			char[] s = x.toCharArray();
			int lt = 0, rt = x.length() - 1;
			while(lt<rt) {
				char tmp = s[lt];
				s[lt] = s[rt];
				s[rt] = tmp;
				lt++;
				rt--;
			}
			String tmp = String.valueOf(s); //String.valueOf(); //스트링 스태틱 선언된 클래스 메서드
			answer.add(tmp);
			
			// 0 1 2 3 4
			// S T U D Y
			
			// lt = 0 , rt = x.length - 1
			// lt는 한개씩 증가, rt 는 한개씩 감소
			// while(lt < rt)
			
			
		}
		
		return answer;
	}
	
	public static void main(String args[]) {
		Main t = new Main();
		
		Scanner scan = new Scanner(System.in);
		
		int num = scan.nextInt(); 
		
		String[] strArr = new String[num];
		
		for(int i = 0; i < num; i++) {
			strArr[i] = scan.next();
		}
		
		for(String x : t.solution4(num, strArr)) {
			System.out.println(x);
		}
	}
}
